12   def parseTasks(node: Node, validPhysicalResources: List[PhysicalIn], humanResourcesNode: Node): Result[List[Task]] =
13     val validHumanResources = HumanParser.parseHumanResources(humanResourcesNode, validPhysicalResources.map(_.`type`).toSet).getOrElse(List.empty)
14 
15     val taskNodes = node \ "Task"
16     val parsedTasks = traverse(taskNodes, node => parseTask(node, validPhysicalResources, validHumanResources))
17 
18     parsedTasks.flatMap { tasks =>
19       val taskWithMissingResources = tasks.find(task =>
20         task.physicalResources.exists(reqType =>
21           validPhysicalResources.count(_.`type` == reqType.`type`) < task.physicalResources.count(_.`type` == reqType.`type`)
22         )
23       )
24 
25       val taskWithMissingHumans = tasks.find(task =>
26         task.physicalResources.exists(reqType =>
27           validHumanResources.count(_.handles.exists(_.`type` == reqType.`type`)) < task.physicalResources.count(_.`type` == reqType.`type`)
28         )
29       )
30 
31       (taskWithMissingResources, taskWithMissingHumans) match
32         case (Some(task), _) => Left(ResourceUnavailable(task.id.id, task.physicalResources.map(_.`type`.`type`).distinct.headOption.getOrElse("Unknown")))
33         case (_, Some(task)) => Left(ResourceUnavailable(task.id.id, task.physicalResources.map(_.`type`.`type`).distinct.headOption.getOrElse("Unknown")))
34         case _              => Right(tasks)
35     }

37   private def parseTask(node: Node, validPhysicalResources: List[PhysicalIn], validHumanResources: List[HumanIn]): Result[Task] =
38     fromAttribute(node, "id").flatMap { idStr =>
39       if (!idStr.startsWith("TSK_"))
40         Left(InvalidTaskId(idStr))
41       else
42         fromAttribute(node, "time").flatMap { timeStr =>
43           val time = Try(timeStr.toInt).toOption
44           time match
45             case Some(parsedTime) =>
46               val physicalResourceNodes = node \ "PhysicalResource"
47               traverse(physicalResourceNodes, pr => validatePhysicalResourceType(pr, validPhysicalResources)).flatMap { physicalResources =>
48 
49                 // Verifica disponibilidade de PhysicalResources
50                 val requiredTypes = physicalResources.map(_.`type`)
51                 val availablePhysicals = validPhysicalResources.filter(p => requiredTypes.contains(p.`type`))
52 
53                 if (availablePhysicals.sizeIs < physicalResources.size)
54                   Left(ResourceUnavailable(idStr, requiredTypes.map(_.`type`).distinct.headOption.getOrElse("Unknown")))
55                 else
56 
57                   // Valida disponibilidade de HumanResources
58                   val requiredHumans = requiredTypes.flatMap(t => validHumanResources.find(_.handles.exists(_.`type` == t)))
59                   val availableHumans = requiredHumans.size
60 
61                   if (availableHumans < physicalResources.size)
62                     Left(ResourceUnavailable(idStr, requiredTypes.map(_.`type`).distinct.headOption.getOrElse("Unknown")))
63                   else
64                     println(s"Task: $idStr, $parsedTime, $physicalResources")
65                     Right(Task(TaskId(idStr), TaskTime(parsedTime), physicalResources))
66               }
67             case None =>
68               Left(IOFileProblem(s"InvalidTimeValue($timeStr)"))
69         }
70     }

72   private def validatePhysicalResourceType(node: Node, validPhysicalResources: List[PhysicalIn]): Result[PhysicalResource] =
73     fromAttribute(node, "type").flatMap { resourceType =>
74       val resource = PhysicalResource(PhysicalInType(resourceType))
75       if (validPhysicalResources.exists(_.`type` == resource.`type`))
76         Right(resource)
77       else
78         Left(TaskUsesNonExistentPRT(resourceType))
79     }
